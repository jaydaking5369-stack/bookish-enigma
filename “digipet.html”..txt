  <button onclick="feed()">Feed (Pepper)</button>

  <button onclick="fight()">Fight Wild Digimon!</button>

<script>
let hunger = 50, happy = 50, level = 1;
let name = "Agumon";
let evolved = false;

function update() {
  document.getElementById('status').innerText = `Hunger: ${hunger} | Happiness: ${happy} | Level: ${level}${evolved ? " (Greymon!)" : ""}`;
  if (hunger > 80 || happy < 20) {
    document.getElementById('message').innerText = `${name} is sad... might run away!`;
  } else {
    document.getElementById('message').innerText = "";
  }
  // Simple evolve
  if (happy > 90 && level === 1) {
    level = 2;
    evolved = true;
    document.getElementById('message').innerText = "Agumon digivolved to Greymon!";
    // Swap sprite if you want (find Greymon pixel art online)
  }
}

function feed() {
  hunger = Math.max(0, hunger - 30);
  happy += 10;
  document.getElementById('message').innerText = "Yum! Pepper time!";
  update();
}

function play() {
  happy += 20;
  hunger += 15;
  document.getElementById('message').innerText = "Woo! Tag—you're it!";
  update();
}

function train() {
  happy -= 10;
  hunger += 10;
  level += 0.5;
  document.getElementById('message').innerText = "Training hard... stronger now!";
  update();
}

function fight() {
  if (hunger > 70 || happy < 30) {
    document.getElementById('message').innerText = "Too weak! Agumon flees...";
    return;
  }
  let win = Math.random() > 0.4; // 60% chance
  if (win) {
    document.getElementById('message').innerText = "Mega Flame! You win—+20 happy!";
    happy += 20;
  } else {
    document.getElementById('message').innerText = "Ouch! Lost—hunger up!";
    hunger += 30;
  }
  update();
}

// Auto-decrease every 10s (like real Tamagotchi)
setInterval(() => {
  hunger += 5;
  happy -= 3;
  update();
}, 10000);

update(); // Start
</script>
